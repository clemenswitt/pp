=======DTD Datei (Datentypen)

<!ELEMENT root (element1, element2)>
<!ELEMENT element1 (subelement1, subelement2)>
<!ELEMENT element2 (subelement3)>

<!ELEMENT subelement1 (#PCDATA)>
<!ELEMENT subelement2 (#PCDATA)>
<!ELEMENT subelement3 (#PCDATA)>

=======XML Datei (Instanzen)

<?xml version="1.0"?>
<!DOCTYPE root SYSTEM "example.dtd">
<root>
    <element1>
        <subelement1>Value 1</subelement1>
        <subelement2>Value 2</subelement2>
    </element1>
    <element2>
        <subelement3>Value 3</subelement3>
    </element2>
</root>

======XSL- Datei (Transformed XML Document)  

<?xml version="1.0" encoding="UTF-8"?>
<xsl:stylesheet version="1.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform">
    <xsl:template match="/">
        <html>
            <head>
                <title>Transformed Document</title>
            </head>
            <body>
                <h1>Transformed XML Document</h1>
                <xsl:apply-templates select="root"/>
            </body>
        </html>
    </xsl:template>

    <xsl:template match="root">
        <ul>
            <xsl:apply-templates/>
        </ul>
    </xsl:template>

    <xsl:template match="element1 | element2">
        <li>
            <xsl:value-of select="name()"/>:
            <ul>
                <xsl:apply-templates/>
            </ul>
        </li>
    </xsl:template>

    <xsl:template match="subelement1 | subelement2 | subelement3">
        <li>
            <xsl:value-of select="name()"/>: <xsl:value-of select="."/>
        </li>
    </xsl:template>
</xsl:stylesheet>
